def _section_header(st, title: str, badge: str | None = None, link: str | None = None) -> None:
    right = f"<span class='sb-section-badge'>{badge}</span>" if badge else \
            (f"<span class='sb-section-link'>{link}</span>" if link else "")
    st.sidebar.markdown(
        f"""
        <div class='sb-section'>
          <div class='sb-section-title'>{title}</div>
          {right}
        </div>
        """,
        unsafe_allow_html=True,
    )

def _preset_stack(st, current: str, options: list[str]) -> str:
    # Use st.radio com label_visibility="collapsed" para manter acessibilidade,
    # mas desenhe o visual com markdown ANTES — e esconda o widget visualmente:
    html = ["<div class='preset-stack'>"]
    for i, name in enumerate(options, start=1):
        active = "is-active" if name == current else ""
        html.append(
            f"<button class='preset-item {active}'>"
            f"<span class='dot'></span><span>{name}</span>"
            f"<span class='cmd'>{i}</span></button>"
        )
    html.append("</div>")
    st.sidebar.markdown("".join(html), unsafe_allow_html=True)
    return st.sidebar.radio(
        "Preset", options, index=options.index(current),
        label_visibility="collapsed", key="_preset_radio",
    )

def _bubble_open(role: str, name: str, badges: list[str], time_str: str = "") -> str:
    avatar = "VC" if role == "user" else "A"
    badges_html = "".join(f"<span class='badge'>{b}</span>" for b in badges)
    return f"""
    <div class='msg {role}'>
